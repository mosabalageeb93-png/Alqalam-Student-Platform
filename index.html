async function login() { const user = document.getElementById('username').value; const pass = document.getElementById('password').value; const uiLoginBtn = document.getElementById('ui-login'); // تغيير شكل الزر أثناء التحميل uiLoginBtn.innerText = currentLang === 'ar' ? "جاري التحقق..." : "Verifying..."; try { // إرسال البيانات لسيرفر البايثون في Termux const response = await fetch('http://10.186.96.68:5000/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: user, password: pass }) }); const result = await response.json(); if (response.ok) { // حفظ البيانات القادمة من قاعدة البيانات localStorage.setItem('isLoggedIn', 'true'); localStorage.setItem('user_real_name', result.name); localStorage.setItem('student_id', result.student_id); localStorage.setItem('batch', result.batch); // الانتقال للوحة التحكم window.location.href = 'dashboard.html'; } else { alert(currentLang === 'ar' ? result.message : "Login Failed"); uiLoginBtn.innerText = currentLang === 'ar' ? "دخول للمنصة" : "Access Platform"; } } catch (error) { console.error('Error:', error); alert(currentLang === 'ar' ? "فشل الاتصال بالسيرفر! تأكد أن Termux يعمل" : "Server Connection Failed!"); uiLoginBtn.innerText = currentLang === 'ar' ? "دخول للمنصة" : "Access Platform"; } }
